class bank_main_kdTree{
    Node root;
    int size;
    public void insert(Node point)
    {
        root=insert(root,point,0);
    }
    public Node insert(Node root,Node point,int depth)
    {
        if (root==null)
        {
            root=new Node(point.bank_main, point.bank_branch, point.coordinates);
            return root;
        }
        int cd=depth%2;
        if (point.coordinates[cd] < root.coordinates[cd])
        {
            insert(root.left,point,depth+1);
        }
        else
        {
            insert(root.right,point,depth+1);
        }
        return root;
    }
    public void print_all()
    {
        print_all(root);
    }
    public void print_all(Node current)
    {
        if (current==null)
        {
            return;
        }
        print_all(current.left);
        System.out.println(current.coordinates[0]+" "+current.coordinates[1]);
        print_all(current.right);
    }
    public Node nearest(Node target)
    {
        return neareset(root,target,0);
    }
    public Node neareset(Node root,Node target,int depth)
    {
        if (root==null)
            return null;
            Node nextBranch = null;
            Node otherBranch = null;
        if (target.coordinates < root.point.get(depth)) {
            nextBranch = root.left;
            otherBranch = root.right;
        } else {
            nextBranch = root.right;
            otherBranch = root.left;
        }
    }

}
